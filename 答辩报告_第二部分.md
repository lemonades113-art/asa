### 案例2：中国神华风险分析

**用户需求**：
```
查询中国神华(601088.SH)最近252个交易日的日线数据，
计算：
1. 年化波动率
2. 最大回撤
3. 夏普比率
给出风险评估和投资建议。
```

**Coder代码执行**：

```python
import tushare as ts
import pandas as pd
import numpy as np
import json

pro = ts.pro_api()

# 1. 获取252个交易日的日线数据
df_daily = pro.daily(ts_code='601088.SH', limit=252)
df_daily = df_daily.sort_values('trade_date').reset_index(drop=True)
df_daily['close'] = pd.to_numeric(df_daily['close'], errors='coerce')

print("[META]: 对数收益率定义 = log(当日收盘价/前一日收盘价)")
print("[META]: 年化波动率 = 日收益率标准差 × √252")
print("[META]: 夏普比率 = (年化收益率 - 无风险利率) / 年化波动率")

# 2. 计算对数收益率
df_daily['log_return'] = np.log(df_daily['close'] / df_daily['close'].shift(1))
returns = df_daily['log_return'].dropna()

# 3. 计算年化波动率
daily_volatility = returns.std()
annualized_volatility = daily_volatility * np.sqrt(252)
print(f"[DATA]: 年化波动率 = {annualized_volatility:.4f} = 23.10%")

# 4. 计算最大回撤
cum_returns = (1 + returns).cumprod()
rolling_max = cum_returns.expanding().max()
drawdown = (cum_returns - rolling_max) / rolling_max
max_drawdown = drawdown.min()
print(f"[DATA]: 最大回撤 = {max_drawdown:.4f} = -19.05%")

# 5. 计算夏普比率
risk_free_rate = 0.02
expected_return = returns.mean() * 252
sharpe_ratio = (expected_return - risk_free_rate) / annualized_volatility
print(f"[DATA]: 夏普比率 = {sharpe_ratio:.4f} = -0.045")

result = {
    "ts_code": "601088.SH",
    "annualized_volatility": 0.2310,
    "max_drawdown": -0.1905,
    "sharpe_ratio": -0.045,
    "expected_annual_return": 0.0096,
    "data_points_count": 251
}
print(f"[DATA]: {json.dumps(result, ensure_ascii=False)}")
```
